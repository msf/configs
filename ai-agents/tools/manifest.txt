# Agent-config manifest. Read by tools/{apply,checkpoint,track}.sh
#
# Format (one entry per line, whitespace-separated, # for comments):
#
#   mirror  <path-rel-to-HOME>
#       → links $HOME/<path>  →  <public>/home/<path>
#   pmirror <path-rel-to-HOME>
#       → links $HOME/<path>  →  ~/configs-private/home/<path>
#       For content with internal endpoints / org identity / infra topology.
#       Filenames appearing here are deliberate; contents stay private.
#   repo    <path-rel-to-HOME>  <target-rel-to-public-repo-root>
#       → links $HOME/<path>  →  <configs-repo>/<target>
#   prepo   <path-rel-to-HOME>  <target-rel-to-private-repo-root>
#       → links $HOME/<path>  →  ~/configs-private/<target>
#   home    <path-rel-to-HOME>  <other-path-rel-to-HOME>
#       → links $HOME/<path>  →  $HOME/<other-path>  (symlink chains)
#
# Order matters: targets referenced through earlier symlinks (e.g. files
# placed *inside* a linked dir) must come after the entry creating that dir.
